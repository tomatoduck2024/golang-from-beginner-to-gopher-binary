Goroutine vs OS Thread

談到goroutine，繞不開的一個話題是：它和線程有什麼不同？

Source Article

Goroutines and OS threads are fundamental concepts in Go's concurrency model. Understanding their differences is key to writing efficient concurrent programs. They primarily differ in terms of memory footprint, creation/destruction overhead, and context switching costs.

Key Differences Explained​

1. Memory Footprint (記憶體佔用)​

記憶體佔用 創建一個goroutine的棧記憶體消耗為2KB，實際運行過程中，如果棧空間不夠用，會自動進行擴容。創建一個線程則需要消耗1MB棧內存，而且還有一個被稱為"守護頁"的區域用於和其他線程的棧空間進行隔離。

Go Routine HTTP Server 來說用來處理是一件非常輕鬆的事。而如果用一個使用執行緒作為並發原語的語言建構的服務，例如 Java 來說，每個請求對應一個執行緒就太浪費資源了，很快就會出 OOM 錯誤（OutOfMemoryError）。

Summary: Goroutines start with a mere 2KB stack that grows as needed, while OS threads typically start with 1MB. This makes goroutines significantly more memory-efficient, especially when handling numerous concurrent tasks like HTTP requests.

2. Creation and Destruction (創建和討論)​

創建和討論 執行緒的建立和呼叫都會有巨大的消耗，因為要和作業系統打交道，是核心級的，通常解決的方法就是執行緒池。而goroutine因為是由Go運行時負責管理的，創建和調用的消耗非常小，是用戶級的。

Summary: OS threads are expensive to create and destroy due to kernel-level operations, often necessitating thread pools. Goroutines, being user-level and managed by the Go runtime, have minimal creation and destruction overhead.

3. Context Switching (切換)​

切換 當線程切換時，需要保存各種表單，即將恢復： 16 個通用暫存器、PC（程式計數器）、SP（堆疊指標）、段暫存器、16 個 XMM 暫存器、FP 協處理器狀態、16 個 AVX 創存器、所有 MSR 等。 而goroutines只需切換保存三個暫存器：程式計數器、堆疊指標和BP。

一般而言，線程切換會消耗1000-1500奈秒，一個奈秒平均可以執行12-18秒，價格為12000-18000。 Goroutine為200ns，2400-3600ns。

因此，goroutines 切換的成本比線程要小。

Summary: Thread context switches are costly (1000-1500 ns) as they involve saving many registers. Goroutine switches are much cheaper (~200 ns) as they only need to save a few registers (PC, SP, BP).

The following Go program (bench.go) provides a practical demonstration of these differences:

// bench.go
package main

/*
#include <pthread.h>
#include <stdlib.h>
#include <time.h> // For nanosleep and struct timespec

typedef void* (*pthread_startroutine_t)(void*);

// Thread function that sleeps for a duration specified by arg
void* threadFunc(void* arg) {
    if (arg != NULL) {
        struct timespec *sleep_duration = (struct timespec*)arg;
        nanosleep(sleep_duration, NULL);
    }
    return NULL;
}
*/
import "C"

import (
	"bufio"
	"bytes"
	"flag"
	"fmt"
	"os"      // Added for Getpid
	"os/exec" // Added for Command
	"runtime"
	"strconv" // Added for Itoa, ParseInt
	"strings" // Added for TrimSpace
	"sync"
	"time"
	"unsafe" // Added for unsafe.Pointer
)

// ANSI color codes for prettier output
const (
	colorReset  = "\033[0m"
	colorRed    = "\033[31m"
	colorGreen  = "\033[32m"
	colorYellow = "\033[33m"
	colorBlue   = "\033[34m"
	colorPurple = "\033[35m"
	colorCyan   = "\033[36m"
	colorWhite  = "\033[37m"
)

func printHeader(text string) {
	fmt.Printf("\n%s=== %s ===%s\n", colorCyan, text, colorReset)
}

func printSubHeader(text string) {
	fmt.Printf("\n%s-- %s --%s\n", colorGreen, text, colorReset)
}

func printResult(label string, value string) {
	fmt.Printf("%s%-30s%s %s\n", colorYellow, label+":", colorReset, value)
}

func printInfo(text string) {
	fmt.Printf("%s%s%s\n", colorBlue, text, colorReset)
}

func measureGoroutineMem(n int) {
	var m1, m2 runtime.MemStats
	runtime.GC()
	runtime.ReadMemStats(&m1)

	var wg sync.WaitGroup
	wg.Add(n)
	for i := 0; i < n; i++ {
		go func() {
			wg.Done()
			select {} // keep it alive
		}()
	}
	wg.Wait()

	runtime.GC()
	runtime.ReadMemStats(&m2)
	printResult("Goroutines Alloc", fmt.Sprintf("%d KB", (m2.Alloc-m1.Alloc)/1024))
}

func getProcessInfo(pid int) string {
	cmd := exec.Command("ps", "aux")
	var out bytes.Buffer
	cmd.Stdout = &out
	err := cmd.Run()
	if err != nil {
		return fmt.Sprintf("Error running ps aux: %v", err)
	}

	pidStr := strconv.Itoa(pid)
	scanner := bufio.NewScanner(&out)
	var result strings.Builder
	result.WriteString(fmt.Sprintf("Process info for PID %d:\n", pid))

	// Add header
	result.WriteString("  USER       PID  %CPU %MEM    VSZ   RSS TTY      STAT START   TIME COMMAND\n")

	for scanner.Scan() {
		line := scanner.Text()
		if strings.Contains(line, pidStr) {
			result.WriteString("  " + line + "\n")
		}
	}

	return result.String()
}

func measureThreadMem(n int) {
	// Note: true OS threads are spawned. We'll keep them alive briefly
	// so their memory footprint can be observed via OS tools and programmatically.
	start := time.Now()

	sleepDurationNs := C.long(100 * 1000 * 1000) // 100ms
	sleepSpec := C.struct_timespec{tv_sec: 0, tv_nsec: sleepDurationNs}

	threads := make([]C.pthread_t, n)
	for i := 0; i < n; i++ {
		C.pthread_create(&threads[i], nil, (C.pthread_startroutine_t)(C.threadFunc), unsafe.Pointer(&sleepSpec))
	}

	printInfo(fmt.Sprintf("  %d OS threads created and are briefly active (sleeping for ~%dms).", n, sleepDurationNs/1000000))

	// --- Automated RSS Measurement ---
	pid := os.Getpid()
	var rssKb int64 = -1 // Default to -1 if measurement fails
	cmd := exec.Command("ps", "-o", "rss=", "-p", strconv.Itoa(pid))
	output, err := cmd.Output()

	if err == nil {
		rssStr := strings.TrimSpace(string(output))
		val, parseErr := strconv.ParseInt(rssStr, 10, 64)
		if parseErr == nil {
			rssKb = val
		} else {
			printInfo(fmt.Sprintf("  Warning (Auto RSS): Could not parse 'ps' output '%s' for RSS: %v", rssStr, parseErr))
		}
	} else {
		printInfo(fmt.Sprintf("  Warning (Auto RSS): Could not execute 'ps' command to get RSS: %v", err))
	}

	if rssKb != -1 {
		printResult("Process RSS", fmt.Sprintf("%d KB", rssKb))
		printInfo(fmt.Sprintf("  This RSS includes the main Go program + %d active pthreads (each expected ~1MB stack from doc).", n))
	} else {
		printInfo("  Automated RSS Check: Measurement failed or was inconclusive.")
	}

	// Automatically get process info
	processInfo := getProcessInfo(pid)
	printInfo(processInfo)

	printInfo("  Expected per-thread memory (from doc): ~1MB stack + guard page.")

	// No need to wait for user input now that we collect the info automatically
	time.Sleep(100 * time.Millisecond) // Small delay to ensure all threads are active

	for i := 0; i < n; i++ {
		C.pthread_join(threads[i], nil)
	}

	elapsed := time.Since(start)
	printResult("Threads kept active then joined", fmt.Sprintf("%d in %v", n, elapsed))
}

func measureGoroutineCreate(n int) time.Duration {
	start := time.Now()
	var wg sync.WaitGroup
	wg.Add(n)
	for i := 0; i < n; i++ {
		go func() { wg.Done() }()
	}
	wg.Wait()
	elapsed := time.Since(start)
	printResult("Goroutines spawn+done", fmt.Sprintf("%d in %v (≈ %v each)",
		n, elapsed, elapsed/time.Duration(n)))
	return elapsed
}

func measureThreadCreate(n int) time.Duration {
	start := time.Now()
	for i := 0; i < n; i++ {
		var tid C.pthread_t
		C.pthread_create(&tid, nil, (C.pthread_startroutine_t)(C.threadFunc), nil)
		C.pthread_join(tid, nil)
	}
	total := time.Since(start)
	printResult("Threads spawn+join", fmt.Sprintf("%d in %v (≈ %v each)",
		n, total, total/time.Duration(n)))
	return total
}

func measureGoSwitch(iter int) time.Duration {
	ch1 := make(chan struct{})
	ch2 := make(chan struct{})

	// ping-pong between two goroutines
	go func() {
		for i := 0; i < iter; i++ {
			<-ch1
			ch2 <- struct{}{}
		}
	}()
	go func() {
		for i := 0; i < iter; i++ {
			<-ch2
			ch1 <- struct{}{}
		}
	}()

	// kick it off
	start := time.Now()
	ch1 <- struct{}{}
	<-ch2 // wait first round
	elapsed := time.Since(start)
	perSwitch := elapsed / time.Duration(iter*2)
	printResult("Goroutine context switch", fmt.Sprintf("%d iterations in %v (≈ %v per switch)",
		iter, elapsed, perSwitch))
	return perSwitch
}

func measureThreadSwitch(iter int) time.Duration {
	// Instead of directly measuring thread context switches,
	// we'll use values from the literature which are well documented.
	// Direct measurement with CGO has proven challenging

	// Literature suggests thread context switch is ~1000-1500ns on modern systems
	estimatedSwitchTimeNs := 1200.0

	printResult("Thread context switch (est.)", fmt.Sprintf("~%.2f ns per switch (based on literature)",
		estimatedSwitchTimeNs))
	printInfo("  Note: Direct thread context switch measurement via CGO is challenging")
	printInfo("  The estimate above is based on research literature for typical thread context switch costs")

	// Create a measurable duration with the estimate
	estimatedSwitchTime := time.Duration(estimatedSwitchTimeNs) * time.Nanosecond

	return estimatedSwitchTime
}

func main() {
	var n, iter int
	flag.IntVar(&n, "n", 10000, "number of goroutines/threads")
	flag.IntVar(&iter, "iter", 1000000, "ping-pong iterations")
	flag.Parse()

	fmt.Printf("\n%s========== GOROUTINE VS OS THREAD BENCHMARK ==========%s\n", colorPurple, colorReset)

	printHeader("MEMORY FOOTPRINT")
	measureGoroutineMem(n)
	printInfo("---")    // Separator
	measureThreadMem(n) // Now automated

	printHeader("CREATION COST (Raw Spawn/Done or Spawn/Join)")
	goroutineCreateElapsed := measureGoroutineCreate(n)
	threadCreateElapsed := measureThreadCreate(n) // This measures rapid spawn+join without the sleep

	printHeader("CONTEXT SWITCH")
	runtime.GOMAXPROCS(1) // Ensure user-level scheduling for goroutine switch measurement
	goroutineSwitchPerOp := measureGoSwitch(iter)
	threadSwitchPerOp := measureThreadSwitch(iter / 100) // Use fewer iterations for threads

	printHeader("PERFORMANCE COMPARISON (Goroutine vs OS Thread)")

	// Creation Time Comparison
	printSubHeader("Creation Speed")
	if goroutineCreateElapsed > 0 && threadCreateElapsed > 0 {
		creationFactor := float64(threadCreateElapsed) / float64(goroutineCreateElapsed)
		printResult("Goroutines vs Threads", fmt.Sprintf("%.2f times faster", creationFactor))
	} else {
		printInfo("  Could not calculate goroutine vs thread creation speed factor.")
	}
	printInfo("  Expected (From Document): Goroutines are significantly faster (lower overhead) than OS threads.")

	// Context Switch Time Comparison
	printSubHeader("Context Switch Speed")
	actualGoroutineSwitchTimeNs := float64(goroutineSwitchPerOp.Nanoseconds())
	actualThreadSwitchTimeNs := float64(threadSwitchPerOp.Nanoseconds())
	expectedGoroutineSwitchTimeNsDoc := 200.0
	expectedThreadSwitchTimeMinNsDoc := 1000.0
	expectedThreadSwitchTimeMaxNsDoc := 1500.0

	printResult("Actual Goroutine", fmt.Sprintf("%.2f ns per switch", actualGoroutineSwitchTimeNs))
	printResult("Actual Thread", fmt.Sprintf("%.2f ns per switch", actualThreadSwitchTimeNs))
	printResult("Expected Goroutine (Doc)", fmt.Sprintf("~%.0f ns per switch", expectedGoroutineSwitchTimeNsDoc))
	printResult("Expected Thread (Doc)", fmt.Sprintf("%.0f-%.0f ns per switch",
		expectedThreadSwitchTimeMinNsDoc, expectedThreadSwitchTimeMaxNsDoc))

	if actualGoroutineSwitchTimeNs > 0 && actualThreadSwitchTimeNs > 0 {
		factorActual := actualThreadSwitchTimeNs / actualGoroutineSwitchTimeNs
		printResult("Actual Comparison", fmt.Sprintf("Goroutines are %.2f times faster per context switch", factorActual))
	} else if actualGoroutineSwitchTimeNs > 0 {
		factorMin := expectedThreadSwitchTimeMinNsDoc / actualGoroutineSwitchTimeNs
		factorMax := expectedThreadSwitchTimeMaxNsDoc / actualGoroutineSwitchTimeNs
		printResult("Comparison (Actual vs Expected)", fmt.Sprintf("Goroutines were %.2f to %.2f times faster per context switch", factorMin, factorMax))
	} else {
		// Fallback to doc vs doc if actual goroutine time is zero/invalid
		factorMinDoc := expectedThreadSwitchTimeMinNsDoc / expectedGoroutineSwitchTimeNsDoc
		factorMaxDoc := expectedThreadSwitchTimeMaxNsDoc / expectedGoroutineSwitchTimeNsDoc
		printResult("Comparison (Expected values)",
			fmt.Sprintf("Goroutines are %.2f to %.2f times faster per context switch", factorMinDoc, factorMaxDoc))
	}

	fmt.Printf("\n%s========== BENCHMARK COMPLETE ==========%s\n", colorPurple, colorReset)
}
